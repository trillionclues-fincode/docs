---
title: 'Get Eligibility Questions'
api: 'GET https://remitjunction.fincode.software/api/v1/services/user/loan-eligibility-questions/{loan-product-id}'
description: 'Retrieves the eligibility questions required for a specific loan product.'
---

Fetches all eligibility questions that customers need to answer when applying for a specific loan product. These questions are used to determine if the customer meets the product's eligibility criteria.

<RequestExample>
  ```bash cURL GET
  [https://remitjunction.fincode.software/api/v1/services/user/loan-eligibility-questions/{loan-product-id}](https://remitjunction.fincode.software/api/v1/services/user/loan-eligibility-questions/{loan-product-id})
  ```
</RequestExample>

## Request Headers

<ParamField header='X-Auth-Token' type='string' required>
  The **JWT Access Token** obtained from the `/login` or `/refresh-token` endpoint.
</ParamField>

<ParamField header='x-fapi-auth-date' type='string' required>
  The date and time at which the request was initiated (ISO 8601 format).
</ParamField>

<ParamField header='x-fapi-customer-ip-address' type='string' required>
  The IP address of the customer making the request.
</ParamField>

<ParamField header='x-fapi-interaction-id' type='string' required>
  Unique identifier for the interaction/session.
</ParamField>

## Path Parameters

<ParamField path='loan-product-id' type='string' required>
  The unique identifier (UUID) of the loan product whose eligibility questions to retrieve.
</ParamField>

## Response

Returns a list of eligibility questions for the loan product.

<ResponseField name='status' type='string' default='SUCCESS'>
  Overall status of the API request.
</ResponseField>

<ResponseField name='data' type='array'>
  Array of eligibility question objects.
  <Expandable title='data array item'>
    <ResponseField name='questionId' type='string'>
      Unique identifier for the question.
    </ResponseField>
    <ResponseField name='questionText' type='string'>
      The question text to display to the customer.
    </ResponseField>
    <ResponseField name='questionType' type='string'>
      Type of question (e.g., `TEXT`, `NUMBER`, `BOOLEAN`, `SELECT`, `DATE`).
    </ResponseField>
    <ResponseField name='isRequired' type='boolean'>
      Whether this question is mandatory.
    </ResponseField>
    <ResponseField name='options' type='array'>
      Available options for select-type questions.
    </ResponseField>
    <ResponseField name='validationRules' type='object'>
      Validation rules for the answer (e.g., min/max values, format requirements).
    </ResponseField>
  </Expandable>
</ResponseField>

```javascript Node.js
const axios = require('axios');

const BASE_URL = 'https://remitjunction.fincode.software/api/v1/services/user';

async function getEligibilityQuestions(accessToken, loanProductId) {
  try {
    const response = await axios.get(
      `${BASE_URL}/loan-eligibility-questions/${loanProductId}`,
      {
        headers: {
          'X-Auth-Token': accessToken,
          'x-fapi-auth-date': new Date().toISOString(),
          'x-fapi-customer-ip-address': '192.168.1.1',
          'x-fapi-interaction-id': `interaction-${Date.now()}`,
        },
      }
    );

    const questions = response.data.data;
    console.log(`Found ${questions.length} eligibility questions`);
    
    questions.forEach((question, index) => {
      console.log(`${index + 1}. ${question.questionText} (${question.questionType})`);
      if (question.isRequired) {
        console.log('   Required: Yes');
      }
    });

    return questions;
  } catch (error) {
    console.error(
      'Failed to fetch eligibility questions:',
      error.response?.data || error.message
    );
    throw error;
  }
}

// Usage Example
const accessToken = 'YOUR_JWT_ACCESS_TOKEN';
const loanProductId = '123e4567-e89b-12d3-a456-426614174000';
getEligibilityQuestions(accessToken, loanProductId);
```
