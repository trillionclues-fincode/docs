---
title: 'Disburse Loan'
api: 'GET https://remitjunction.fincode.software/api/v6/services/admin/loans/disburse-loan'
description: 'Disburses an approved loan to the customer wallet or bank account.'
---

Disburses funds for an approved loan application. The loan amount is transferred to the customer's designated wallet or bank account based on the disbursement method specified.

<RequestExample>
  ```bash cURL GET
  [https://remitjunction.fincode.software/api/v6/services/admin/loans/disburse-loan](https://remitjunction.fincode.software/api/v6/services/admin/loans/disburse-loan)
  ```
</RequestExample>

<Note>
  This endpoint uses GET method but requires a request body. This is based on the controller implementation. In production, consider using POST for better RESTful practices.
</Note>

## Request Headers

<ParamField header='X-Auth-Token' type='string' required>
  The **JWT Access Token** obtained from the `/login` or `/refresh-token` endpoint.
</ParamField>

<ParamField header='x-idempotency-key' type='string' required>
  Unique idempotency key for the request to prevent duplicate processing.
</ParamField>

<ParamField header='x-fapi-auth-date' type='string' required>
  The date and time at which the request was initiated (ISO 8601 format).
</ParamField>

<ParamField header='x-fapi-customer-ip-address' type='string' required>
  The IP address of the customer making the request.
</ParamField>

<ParamField header='x-fapi-interaction-id' type='string' required>
  Unique identifier for the interaction/session.
</ParamField>

<ParamField header='Content-Type' type='string' required default='application/json'>
  Must be `application/json`.
</ParamField>

## Request Body

<ParamField body='loanApplicationId' type='string' required>
  The unique identifier (UUID) of the approved loan application to disburse.
</ParamField>

<ParamField body='disbursementMethod' type='string' required>
  Method of disbursement. Options: `WALLET`, `BANK_ACCOUNT`, `CASH`.
</ParamField>

<ParamField body='accountNumber' type='string'>
  Bank account number (required if disbursementMethod is `BANK_ACCOUNT`).
</ParamField>

<ParamField body='bankCode' type='string'>
  Bank code (required if disbursementMethod is `BANK_ACCOUNT`).
</ParamField>

<ParamField body='disbursementNotes' type='string'>
  Optional notes or comments about the disbursement.
</ParamField>

## Response

Returns confirmation of the loan disbursement with transaction details.

<ResponseField name='status' type='string' default='SUCCESS'>
  Overall status of the API request.
</ResponseField>

<ResponseField name='data' type='object'>
  Disbursement confirmation details.
  <Expandable title='data object'>
    <ResponseField name='loanApplicationId' type='string'>
      Unique identifier of the loan application.
    </ResponseField>
    <ResponseField name='disbursementId' type='string'>
      Unique identifier for the disbursement transaction.
    </ResponseField>
    <ResponseField name='disbursedAmount' type='number'>
      The amount that was disbursed.
    </ResponseField>
    <ResponseField name='disbursementMethod' type='string'>
      Method used for disbursement.
    </ResponseField>
    <ResponseField name='status' type='string' default='DISBURSED'>
      Updated status of the loan (should be `DISBURSED`).
    </ResponseField>
    <ResponseField name='disbursementDate' type='string'>
      Date and time when the loan was disbursed.
    </ResponseField>
    <ResponseField name='transactionReference' type='string'>
      Reference number for the disbursement transaction.
    </ResponseField>
    <ResponseField name='message' type='string'>
      Confirmation message indicating successful disbursement.
    </ResponseField>
  </Expandable>
</ResponseField>

```javascript Node.js
const axios = require('axios');

const BASE_URL = 'https://remitjunction.fincode.software/api/v6/services/admin/loans';

async function disburseLoan(accessToken, disbursementData) {
  try {
    const response = await axios.get(
      `${BASE_URL}/disburse-loan`,
      {
        data: {
          loanApplicationId: disbursementData.loanApplicationId,
          disbursementMethod: disbursementData.disbursementMethod,
          accountNumber: disbursementData.accountNumber,
          bankCode: disbursementData.bankCode,
          disbursementNotes: disbursementData.disbursementNotes || '',
        },
        headers: {
          'X-Auth-Token': accessToken,
          'x-idempotency-key': `key-${Date.now()}`,
          'x-fapi-auth-date': new Date().toISOString(),
          'x-fapi-customer-ip-address': '192.168.1.1',
          'x-fapi-interaction-id': `interaction-${Date.now()}`,
          'Content-Type': 'application/json',
        },
      }
    );

    console.log('Loan disbursed successfully!');
    console.log('Disbursement ID:', response.data.data.disbursementId);
    console.log('Amount:', response.data.data.disbursedAmount);
    console.log('Transaction Reference:', response.data.data.transactionReference);
    return response.data.data;
  } catch (error) {
    console.error(
      'Failed to disburse loan:',
      error.response?.data || error.message
    );
    throw error;
  }
}

// Usage Example
const accessToken = 'YOUR_JWT_ACCESS_TOKEN';
disburseLoan(accessToken, {
  loanApplicationId: '123e4567-e89b-12d3-a456-426614174000',
  disbursementMethod: 'WALLET',
  disbursementNotes: 'Loan disbursed to customer wallet',
});
```
