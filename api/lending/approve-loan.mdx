---
title: 'Approve Loan'
api: 'PUT https://remitjunction.fincode.software/api/v6/services/admin/loan-application/approve-loan'
description: 'Approves a loan application and sets the approved loan terms including amount, interest rate, and tenure.'
---

Approves a loan application that has been processed and reviewed. This endpoint allows managers or cashiers to approve loan applications with specific terms including approved amount, interest rate, and repayment tenure.

<RequestExample>
  ```bash cURL PUT
  [https://remitjunction.fincode.software/api/v6/services/admin/loan-application/approve-loan](https://remitjunction.fincode.software/api/v6/services/admin/loan-application/approve-loan)
  ```
</RequestExample>

## Request Headers

<ParamField header='X-Auth-Token' type='string' required>
  The **JWT Access Token** obtained from the `/login` or `/refresh-token` endpoint.
</ParamField>

<ParamField header='x-idempotency-key' type='string' required>
  Unique idempotency key for the request to prevent duplicate processing.
</ParamField>

<ParamField header='x-fapi-auth-date' type='string' required>
  The date and time at which the request was initiated (ISO 8601 format).
</ParamField>

<ParamField header='x-fapi-customer-ip-address' type='string' required>
  The IP address of the customer making the request.
</ParamField>

<ParamField header='x-fapi-interaction-id' type='string' required>
  Unique identifier for the interaction/session.
</ParamField>

<ParamField header='Content-Type' type='string' required default='application/json'>
  Must be `application/json`.
</ParamField>

<ParamField header='platform' type='string' required default='fincode'>
  Platform identifier. Use `fincode`.
</ParamField>

<ParamField header='uuid' type='string' required default='200'>
  Unique request identifier.
</ParamField>

## Request Body

<ParamField body='loanApplicationId' type='string' required>
  The unique identifier (UUID) of the loan application to approve.
</ParamField>

<ParamField body='approvedAmount' type='number' required>
  The loan amount being approved (may differ from requested amount).
</ParamField>

<ParamField body='interestRate' type='number' required>
  The annual interest rate for the loan (e.g., 12.5 for 12.5%).
</ParamField>

<ParamField body='tenure' type='number' required>
  Loan tenure in months (e.g., 6, 12, 24).
</ParamField>

<ParamField body='approvalComments' type='string'>
  Optional comments or notes about the approval decision.
</ParamField>

## Response

Returns confirmation of the loan approval with updated application status.

<ResponseField name='status' type='string' default='SUCCESS'>
  Overall status of the API request.
</ResponseField>

<ResponseField name='data' type='object'>
  Approval confirmation details.
  <Expandable title='data object'>
    <ResponseField name='loanApplicationId' type='string'>
      Unique identifier of the approved loan application.
    </ResponseField>
    <ResponseField name='status' type='string' default='APPROVED'>
      Updated status of the application (should be `APPROVED`).
    </ResponseField>
    <ResponseField name='approvedAmount' type='number'>
      The approved loan amount.
    </ResponseField>
    <ResponseField name='message' type='string'>
      Confirmation message indicating successful approval.
    </ResponseField>
    <ResponseField name='approvedDate' type='string'>
      Date and time when the loan was approved.
    </ResponseField>
  </Expandable>
</ResponseField>

```javascript Node.js
const axios = require('axios');

const BASE_URL = 'https://remitjunction.fincode.software/api/v6/services/admin/loan-application';

async function approveLoan(accessToken, approvalData) {
  try {
    const response = await axios.put(
      `${BASE_URL}/approve-loan`,
      {
        loanApplicationId: approvalData.loanApplicationId,
        approvedAmount: approvalData.approvedAmount,
        interestRate: approvalData.interestRate,
        tenure: approvalData.tenure,
        approvalComments: approvalData.approvalComments || '',
      },
      {
        headers: {
          'X-Auth-Token': accessToken,
          'x-idempotency-key': `key-${Date.now()}`,
          'x-fapi-auth-date': new Date().toISOString(),
          'x-fapi-customer-ip-address': '192.168.1.1',
          'x-fapi-interaction-id': `interaction-${Date.now()}`,
          'Content-Type': 'application/json',
          platform: 'fincode',
          uuid: '200',
        },
      }
    );

    console.log('Loan approved successfully!');
    console.log('Application ID:', response.data.data.loanApplicationId);
    console.log('Approved Amount:', response.data.data.approvedAmount);
    return response.data.data;
  } catch (error) {
    console.error(
      'Failed to approve loan:',
      error.response?.data || error.message
    );
    throw error;
  }
}

// Usage Example
const accessToken = 'YOUR_JWT_ACCESS_TOKEN';
approveLoan(accessToken, {
  loanApplicationId: '123e4567-e89b-12d3-a456-426614174000',
  approvedAmount: 50000.00,
  interestRate: 12.5,
  tenure: 12,
  approvalComments: 'Customer meets all eligibility criteria.',
});
```
