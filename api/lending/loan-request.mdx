---
title: 'Request Loan'
api: 'POST https://remitjunction.fincode.software/api/v1/services/user/request-loan'
description: 'Creates a new loan application request from a customer or agent.'
---

Creates a new loan application request. Customers or agents can use this endpoint to submit loan applications with required details including loan amount, product selection, and eligibility information.

<RequestExample>
  ```bash cURL POST
  [https://remitjunction.fincode.software/api/v1/services/user/request-loan](https://remitjunction.fincode.software/api/v1/services/user/request-loan)
  ```
</RequestExample>

## Request Headers

<ParamField header='X-Auth-Token' type='string' required>
  The **JWT Access Token** obtained from the `/login` or `/refresh-token` endpoint.
</ParamField>

<ParamField header='x-idempotency-key' type='string' required>
  Unique idempotency key for the request to prevent duplicate processing.
</ParamField>

<ParamField header='x-fapi-auth-date' type='string' required>
  The date and time at which the request was initiated (ISO 8601 format).
</ParamField>

<ParamField header='x-fapi-customer-ip-address' type='string' required>
  The IP address of the customer making the request.
</ParamField>

<ParamField header='x-fapi-interaction-id' type='string' required>
  Unique identifier for the interaction/session.
</ParamField>

<ParamField header='Content-Type' type='string' required default='application/json'>
  Must be `application/json`.
</ParamField>

## Request Body

<ParamField body='customerId' type='string' required>
  The unique identifier (UUID) of the customer applying for the loan.
</ParamField>

<ParamField body='loanProductId' type='string' required>
  The unique identifier of the loan product being applied for.
</ParamField>

<ParamField body='requestedAmount' type='number' required>
  The loan amount being requested. Must be within the product's min/max limits.
</ParamField>

<ParamField body='purpose' type='string'>
  Purpose of the loan (e.g., `BUSINESS`, `PERSONAL`, `EDUCATION`, `MEDICAL`).
</ParamField>

<ParamField body='eligibilityAnswers' type='array'>
  Array of answers to eligibility questions required by the loan product.
  <Expandable title='eligibilityAnswers array item'>
    <ParamField body='eligibilityAnswers[].questionId' type='string' required>
      Unique identifier of the eligibility question.
    </ParamField>
    <ParamField body='eligibilityAnswers[].answer' type='string' required>
      Answer to the eligibility question.
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body='tenurePreference' type='number'>
  Preferred loan tenure in months (optional, may be overridden by product terms).
</ParamField>

<ParamField body='additionalNotes' type='string'>
  Optional additional notes or comments about the loan application.
</ParamField>

## Response

Returns confirmation of the loan application creation with application details.

<ResponseField name='status' type='string' default='SUCCESS'>
  Overall status of the API request.
</ResponseField>

<ResponseField name='data' type='object'>
  Loan application response details.
  <Expandable title='data object'>
    <ResponseField name='loanApplicationId' type='string'>
      Unique identifier for the newly created loan application.
    </ResponseField>
    <ResponseField name='loanRequestId' type='string'>
      Unique identifier for the loan request (initial request before processing).
    </ResponseField>
    <ResponseField name='status' type='string' default='REQUESTED'>
      Initial status of the application (typically `REQUESTED`).
    </ResponseField>
    <ResponseField name='requestedAmount' type='number'>
      The loan amount that was requested.
    </ResponseField>
    <ResponseField name='loanProductName' type='string'>
      Name of the loan product applied for.
    </ResponseField>
    <ResponseField name='createdDate' type='string'>
      Date and time when the application was created.
    </ResponseField>
    <ResponseField name='message' type='string'>
      Confirmation message indicating successful application creation.
    </ResponseField>
    <ResponseField name='nextSteps' type='array'>
      Array of next steps or actions required (e.g., document upload, verification).
    </ResponseField>
  </Expandable>
</ResponseField>

```javascript Node.js
const axios = require('axios');

const BASE_URL = 'https://remitjunction.fincode.software/api/v1/services/user';

async function requestLoan(accessToken, loanRequestData) {
  try {
    const response = await axios.post(
      `${BASE_URL}/request-loan`,
      {
        customerId: loanRequestData.customerId,
        loanProductId: loanRequestData.loanProductId,
        requestedAmount: loanRequestData.requestedAmount,
        purpose: loanRequestData.purpose || 'PERSONAL',
        tenurePreference: loanRequestData.tenurePreference,
        eligibilityAnswers: loanRequestData.eligibilityAnswers || [],
        additionalNotes: loanRequestData.additionalNotes || '',
      },
      {
        headers: {
          'X-Auth-Token': accessToken,
          'x-idempotency-key': `key-${Date.now()}`,
          'x-fapi-auth-date': new Date().toISOString(),
          'x-fapi-customer-ip-address': '192.168.1.1',
          'x-fapi-interaction-id': `interaction-${Date.now()}`,
          'Content-Type': 'application/json',
        },
      }
    );

    console.log('Loan application created successfully!');
    console.log('Application ID:', response.data.data.loanApplicationId);
    console.log('Status:', response.data.data.status);
    console.log('Requested Amount:', response.data.data.requestedAmount);
    
    if (response.data.data.nextSteps && response.data.data.nextSteps.length > 0) {
      console.log('\nNext Steps:');
      response.data.data.nextSteps.forEach((step, index) => {
        console.log(`  ${index + 1}. ${step}`);
      });
    }

    return response.data.data;
  } catch (error) {
    console.error(
      'Failed to create loan application:',
      error.response?.data || error.message
    );
    throw error;
  }
}

// Usage Example
const accessToken = 'YOUR_JWT_ACCESS_TOKEN';

  eligibilityAnswers: [
    { questionId: 'q1', answer: 'Yes' },
    { questionId: 'q2', answer: '50000' },
  ],
  additionalNotes: 'Need funds for business expansion',
});
```
