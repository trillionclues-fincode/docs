---
title: 'List Applications'
api: 'GET https://remitjunction.fincode.software/api/v6/services/admin/loan-application/loan-application/get-all'
description: 'Retrieves a paginated list of all loan applications in the system.'
---

Fetches a paginated list of all loan applications, allowing managers to view and manage loan requests. The response includes application details, status, and customer information.

<RequestExample>
  ```bash cURL GET
  [https://remitjunction.fincode.software/api/v6/services/admin/loan-application/loan-application/get-all](https://remitjunction.fincode.software/api/v6/services/admin/loan-application/loan-application/get-all)
  ```
</RequestExample>

## Request Headers

<ParamField header='X-Auth-Token' type='string' required>
  The **JWT Access Token** obtained from the `/login` or `/refresh-token` endpoint.
</ParamField>

<ParamField header='x-idempotency-key' type='string' required>
  Unique idempotency key for the request to prevent duplicate processing.
</ParamField>

<ParamField header='x-fapi-auth-date' type='string' required>
  The date and time at which the request was initiated (ISO 8601 format).
</ParamField>

<ParamField header='x-fapi-customer-ip-address' type='string' required>
  The IP address of the customer making the request.
</ParamField>

<ParamField header='x-fapi-interaction-id' type='string' required>
  Unique identifier for the interaction/session.
</ParamField>

<ParamField header='platform' type='string' required default='fincode'>
  Platform identifier. Use `fincode`.
</ParamField>

<ParamField header='uuid' type='string' required default='200'>
  Unique request identifier.
</ParamField>

## Query Parameters

<ParamField query='page' type='number' default='1'>
  Page number for pagination (starts from 1).
</ParamField>

<ParamField query='size' type='number' default='10'>
  Number of items per page (maximum recommended: 50).
</ParamField>

## Response

Returns a paginated list of loan applications with metadata about total pages and items.

<ResponseField name='status' type='string' default='SUCCESS'>
  Overall status of the API request.
</ResponseField>

<ResponseField name='data' type='object'>

```javascript Node.js
const axios = require('axios');

const BASE_URL = 'https://remitjunction.fincode.software/api/v6/services/admin/loan-application';

async function listLoanApplications(accessToken, page = 1, size = 10) {
  try {
    const response = await axios.get(
      `${BASE_URL}/loan-application/get-all`,
      {
        params: {
          page,
          size,
        },
        headers: {
          'X-Auth-Token': accessToken,
          'x-idempotency-key': `key-${Date.now()}`,
          'x-fapi-auth-date': new Date().toISOString(),
          'x-fapi-customer-ip-address': '192.168.1.1',
          'x-fapi-interaction-id': `interaction-${Date.now()}`,
          platform: 'fincode',
          uuid: '200',
        },
      }
    );

    const applications = response.data.data.content;
    console.log(`Found ${applications.length} loan applications`);
    console.log(`Total: ${response.data.data.totalElements} applications`);
    
    applications.forEach((app) => {
      console.log(`- ${app.loanApplicationId}: ${app.status} - Amount: ${app.requestedAmount}`);
    });

    return response.data.data;
  } catch (error) {
    console.error(
      'Failed to list loan applications:',
      error.response?.data || error.message
    );
    throw error;
  }
}

// Usage Example
const accessToken = 'YOUR_JWT_ACCESS_TOKEN';
listLoanApplications(accessToken, 1, 20);
```
