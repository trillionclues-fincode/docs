---
title: 'Get Loan Details with Payment History'
api: 'GET https://remitjunction.fincode.software/api/v1/services/user/loan-details-with-payment-history/{loan-application-id}'
description: 'Retrieves comprehensive loan details including payment history and installment schedule for a specific loan.'
---

Fetches complete loan information including all payment transactions, installment schedule, due dates, and payment status. This helps customers track their loan progress and upcoming payments.

<RequestExample>
  ```bash cURL GET
  [https://remitjunction.fincode.software/api/v1/services/user/loan-details-with-payment-history/{loan-application-id}](https://remitjunction.fincode.software/api/v1/services/user/loan-details-with-payment-history/{loan-application-id})
  ```
</RequestExample>

## Response

Returns comprehensive loan details including payment history and installment schedule.

<ResponseField name='status' type='string' default='SUCCESS'>
  Overall status of the API request.
</ResponseField>

```javascript Node.js
const axios = require('axios');

const BASE_URL = 'https://remitjunction.fincode.software/api/v1/services/user';

async function getLoanDetailsWithPaymentHistory(accessToken, loanApplicationId) {
  try {
    const response = await axios.get(
      `${BASE_URL}/loan-details-with-payment-history/${loanApplicationId}`,
      {
        headers: {
          'X-Auth-Token': accessToken,
          'x-idempotency-key': `key-${Date.now()}`,
          'x-fapi-auth-date': new Date().toISOString(),
          'x-fapi-customer-ip-address': '192.168.1.1',
          'x-fapi-interaction-id': `interaction-${Date.now()}`,
        },
      }
    );

    const loanDetails = response.data.data;
    console.log('Loan Details:');
    console.log(`Amount: ${loanDetails.loanAmount}`);
    console.log(`Outstanding Balance: ${loanDetails.outstandingBalance}`);
    console.log(`Status: ${loanDetails.status}`);
    console.log(`\nPayment History (${loanDetails.paymentHistory.length} payments):`);
    
    loanDetails.paymentHistory.forEach((payment) => {
      console.log(`  ${payment.paymentDate}: ${payment.paymentAmount} (${payment.paymentMethod})`);
    });

    console.log(`\nInstallments (${loanDetails.installments.length}):`);
    loanDetails.installments.forEach((installment) => {
      console.log(
        `  Installment ${installment.installmentNumber}: ` +
        `${installment.amount} due ${installment.dueDate} - ` +
        `Status: ${installment.status}`
      );
    });

    return loanDetails;
  } catch (error) {
    console.error(
      'Failed to fetch loan details:',
      error.response?.data || error.message
    );
    throw error;
  }
}

// Usage Example
const accessToken = 'YOUR_JWT_ACCESS_TOKEN';
const loanApplicationId = '123e4567-e89b-12d3-a456-426614174000';
getLoanDetailsWithPaymentHistory(accessToken, loanApplicationId);
```
