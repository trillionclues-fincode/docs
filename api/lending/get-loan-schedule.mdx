---
title: 'Get Loan Schedule'
api: 'GET https://remitjunction.fincode.software/api/v6/services/admin/loans/{loan-application-id}/schedule'
description: 'Retrieves the repayment schedule for a loan, showing all scheduled installments and payment dates.'
---

Fetches the complete repayment schedule for a loan, including all installments, due dates, amounts, and payment status. This helps customers and managers track upcoming payments and repayment progress.

<RequestExample>
  ```bash cURL GET
  [https://remitjunction.fincode.software/api/v6/services/admin/loans/{loan-application-id}/schedule](https://remitjunction.fincode.software/api/v6/services/admin/loans/{loan-application-id}/schedule)
  ```
</RequestExample>

## Request Headers

<ParamField header='X-Auth-Token' type='string' required>
  The **JWT Access Token** obtained from the `/login` or `/refresh-token` endpoint.
</ParamField>

<ParamField header='x-idempotency-key' type='string' required>
  Unique idempotency key for the request to prevent duplicate processing.
</ParamField>

<ParamField header='x-fapi-auth-date' type='string' required>
  The date and time at which the request was initiated (ISO 8601 format).
</ParamField>

<ParamField header='x-fapi-customer-ip-address' type='string' required>
  The IP address of the customer making the request.
</ParamField>

<ParamField header='x-fapi-interaction-id' type='string' required>
  Unique identifier for the interaction/session.
</ParamField>

## Path Parameters

<ParamField path='loan-application-id' type='string' required>
  The unique identifier (UUID) of the loan application to retrieve the schedule for.
</ParamField>

Returns the complete repayment schedule with installment details.

```javascript Node.js
const axios = require('axios');

const BASE_URL = 'https://remitjunction.fincode.software/api/v6/services/admin/loans';

async function getLoanSchedule(accessToken, loanApplicationId) {
  try {
    const response = await axios.get(
      `${BASE_URL}/${loanApplicationId}/schedule`,
      {
        headers: {
          'X-Auth-Token': accessToken,
          'x-idempotency-key': `key-${Date.now()}`,
          'x-fapi-auth-date': new Date().toISOString(),
          'x-fapi-customer-ip-address': '192.168.1.1',
          'x-fapi-interaction-id': `interaction-${Date.now()}`,
        },
      }
    );

    const schedule = response.data.data;
    console.log('Loan Schedule:');
    console.log(`Total Amount Due: ${schedule.totalAmountDue}`);
    console.log(`Tenure: ${schedule.tenure} months`);
    console.log(`Interest Rate: ${schedule.interestRate}%`);
    console.log(`\nInstallments (${schedule.installments.length}):`);
    
    schedule.installments.forEach((installment) => {
      console.log(
        `  Installment ${installment.installmentNumber}: ` +
        `${installment.totalAmount} due ${installment.dueDate} - ` +
        `Status: ${installment.status}`
      );
    });

    console.log(`\nSummary:`);
    console.log(`  Paid: ${schedule.summary.totalPaid}`);
    console.log(`  Pending: ${schedule.summary.totalPending}`);
    console.log(`  Paid Installments: ${schedule.summary.paidInstallments}`);
    console.log(`  Pending Installments: ${schedule.summary.pendingInstallments}`);

    return schedule;
  } catch (error) {
    console.error(
      'Failed to fetch loan schedule:',
      error.response?.data || error.message
    );
    throw error;
  }
}

// Usage Example
const accessToken = 'YOUR_JWT_ACCESS_TOKEN';
const loanApplicationId = '123e4567-e89b-12d3-a456-426614174000';
getLoanSchedule(accessToken, loanApplicationId);
```
