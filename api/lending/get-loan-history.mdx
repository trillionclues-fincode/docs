Fetches a comprehensive history of all loans associated with a user, including completed, active, and cancelled loans.

<RequestExample>
  ```bash cURL GET
  [https://remitjunction.fincode.software/api/v1/services/user/get-loan-history/{user-id}](https://remitjunction.fincode.software/api/v1/services/user/get-loan-history/{user-id})
  ```
</RequestExample>

## Request Headers

<ParamField header='X-Auth-Token' type='string' required>
  The **JWT Access Token** obtained from the `/login` or `/refresh-token` endpoint.
</ParamField>

<ParamField header='x-idempotency-key' type='string' required>
  Unique idempotency key for the request to prevent duplicate processing.
</ParamField>

<ParamField header='x-fapi-auth-date' type='string' required>
  The date and time at which the request was initiated (ISO 8601 format).
</ParamField>

<ParamField header='x-fapi-customer-ip-address' type='string' required>
  The IP address of the customer making the request.
</ParamField>

<ParamField header='x-fapi-interaction-id' type='string' required>
  Unique identifier for the interaction/session.
</ParamField>

## Path Parameters

<ParamField path='user-id' type='string' required>
  The unique identifier (UUID) of the user whose loan history to retrieve.
</ParamField>

## Response

Returns a list of all loans in the user's history.

<ResponseField name='status' type='string' default='SUCCESS'>
  Overall status of the API request.
</ResponseField>

<ResponseField name='data' type='array'>
  Array of loan history objects.
  <Expandable title='data array item'>
    <ResponseField name='loanApplicationId' type='string'>
      Unique identifier for the loan application.
    </ResponseField>
    <ResponseField name='loanProductName' type='string'>
      Name of the loan product.
    </ResponseField>
    <ResponseField name='loanAmount' type='number'>
      The loan amount.
    </ResponseField>
    <ResponseField name='status' type='string'>
      Current status of the loan.
    </ResponseField>
    <ResponseField name='applicationDate' type='string'>
      Date when the loan was applied for.
    </ResponseField>
    <ResponseField name='disbursementDate' type='string'>
      Date when the loan was disbursed (if disbursed).
    </ResponseField>
    <ResponseField name='completionDate' type='string'>
      Date when the loan was completed (if completed).
    </ResponseField>
    <ResponseField name='totalRepaid' type='number'>
      Total amount repaid on this loan.
    </ResponseField>
    <ResponseField name='outstandingBalance' type='number'>
      Current outstanding balance (if any).
    </ResponseField>
  </Expandable>
</ResponseField>

```javascript Node.js
const axios = require('axios');

const BASE_URL = 'https://remitjunction.fincode.software/api/v1/services/user';

async function getLoanHistory(accessToken, userId) {
  try {
    const response = await axios.get(
      `${BASE_URL}/get-loan-history/${userId}`,
      {
        headers: {
          'X-Auth-Token': accessToken,
          'x-idempotency-key': `key-${Date.now()}`,
          'x-fapi-auth-date': new Date().toISOString(),
          'x-fapi-customer-ip-address': '192.168.1.1',
          'x-fapi-interaction-id': `interaction-${Date.now()}`,
        },
      }
    );

    const history = response.data.data;
    console.log(`Found ${history.length} loans in history`);
    
    history.forEach((loan) => {
      console.log(
        `- ${loan.loanApplicationId}: ${loan.loanProductName} - ` +
        `${loan.loanAmount} (Status: ${loan.status})`
      );
    });

    return history;
  } catch (error) {
    console.error(
      'Failed to fetch loan history:',
      error.response?.data || error.message
    );
    throw error;
  }
}

// Usage Example
const accessToken = 'YOUR_JWT_ACCESS_TOKEN';
const userId = '123e4567-e89b-12d3-a456-426614174000';
getLoanHistory(accessToken, userId);
```
