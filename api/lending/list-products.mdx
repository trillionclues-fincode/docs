---
title: 'List Loan Products'
api: 'GET https://remitjunction.fincode.software/api/v1/services/user/list-products'
description: 'Retrieves a list of all activated loan products available to customers.'
---

Fetches all active loan products that customers can apply for, including product details, interest rates, tenure options, and eligibility requirements.

<RequestExample>
  ```bash cURL GET
  [https://remitjunction.fincode.software/api/v1/services/user/list-products](https://remitjunction.fincode.software/api/v1/services/user/list-products)
  ```
</RequestExample>

## Request Headers

<ParamField header='X-Auth-Token' type='string' required>
  The **JWT Access Token** obtained from the `/login` or `/refresh-token` endpoint.
</ParamField>

<ParamField header='x-idempotency-key' type='string' required>
  Unique idempotency key for the request to prevent duplicate processing.
</ParamField>

<ParamField header='x-fapi-auth-date' type='string' required>
  The date and time at which the request was initiated (ISO 8601 format).
</ParamField>

<ParamField header='x-fapi-customer-ip-address' type='string' required>
  The IP address of the customer making the request.
</ParamField>

<ParamField header='x-fapi-interaction-id' type='string' required>
  Unique identifier for the interaction/session.
</ParamField>

## Response

Returns a list of all activated loan products.

<ResponseField name='status' type='string' default='SUCCESS'>
  Overall status of the API request.
</ResponseField>

<ResponseField name='data' type='array'>
  Array of loan product objects.
  <Expandable title='data array item'>
    <ResponseField name='loanProductId' type='string'>
      Unique identifier for the loan product.
    </ResponseField>
    <ResponseField name='productName' type='string'>
      Name of the loan product.
    </ResponseField>
    <ResponseField name='description' type='string'>
      Description of the loan product.
    </ResponseField>
    <ResponseField name='minAmount' type='number'>
      Minimum loan amount for this product.
    </ResponseField>
    <ResponseField name='maxAmount' type='number'>
      Maximum loan amount for this product.
    </ResponseField>
    <ResponseField name='interestRate' type='number'>
      Interest rate for this product (annual percentage).
    </ResponseField>
    <ResponseField name='tenureOptions' type='array'>
      Available tenure options in months (e.g., [3, 6, 12, 24]).
    </ResponseField>
    <ResponseField name='eligibilityCriteria' type='object'>
      Eligibility requirements for this product.
    </ResponseField>
    <ResponseField name='isActive' type='boolean'>
      Whether the product is currently active and available.
    </ResponseField>
  </Expandable>
</ResponseField>

```bash cURL
curl --location 'https://remitjunction.fincode.software/api/v1/services/user/list-products' \

```

```javascript Node.js
const axios = require('axios');

const BASE_URL = 'https://remitjunction.fincode.software/api/v1/services/user';

async function listLoanProducts(accessToken) {
  try {
    const response = await axios.get(
      `${BASE_URL}/list-products`,
      {
        headers: {
          'X-Auth-Token': accessToken,
          'x-idempotency-key': `key-${Date.now()}`,
          'x-fapi-auth-date': new Date().toISOString(),
          'x-fapi-customer-ip-address': '192.168.1.1',
          'x-fapi-interaction-id': `interaction-${Date.now()}`,
        },
      }
    );

    const products = response.data.data;
    console.log(`Found ${products.length} loan products`);
    
    products.forEach((product) => {
      console.log(
        `- ${product.productName}: ` +
        `${product.minAmount} - ${product.maxAmount} at ${product.interestRate}%`
      );
    });

    return products;
  } catch (error) {
    console.error(
      'Failed to fetch loan products:',
      error.response?.data || error.message
    );
    throw error;
  }
}

// Usage Example
const accessToken = 'YOUR_JWT_ACCESS_TOKEN';
listLoanProducts(accessToken);
```
