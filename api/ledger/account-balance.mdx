---
title: 'Account Balance'
api: 'GET https://{domain}.fincode.software/api/v6/services/ledger/account-balance/{account_id}'
description: 'Retrieves the current balance and account details for a specific account.'
---

Fetches comprehensive account balance information including available balance, account balance, account status, and account details.

<RequestExample>
  ```bash cURL GET
  [https://{domain}.fincode.software/api/v6/services/ledger/account-balance/{account_id}](https://{domain}.fincode.software/api/v6/services/ledger/account-balance/{account_id})
  ```
</RequestExample>

## Request Headers

<ParamField header='X-Auth-Token' type='string' required>
  The **JWT Access Token** obtained from the `/login` or `/refresh-token` endpoint.
</ParamField>

<ParamField header='x-fapi-auth-date' type='string' required>
  The date and time at which the request was initiated (ISO 8601 format).
</ParamField>

<ParamField header='x-fapi-customer-ip-address' type='string' required>
  The IP address of the customer making the request.
</ParamField>

<ParamField header='x-fapi-interaction-id' type='string' required>
  Unique identifier for the interaction/session.
</ParamField>

## Path Parameters

<ParamField path='account_id' type='string' required>
  The unique identifier (UUID) of the account to query.
</ParamField>

## Response

Returns comprehensive account balance and details.

<ResponseField name='status' type='string' default='SUCCESS'>
  Overall status of the API request.
</ResponseField>

<ResponseField name='data' type='object'>
  Account balance and details.
  <Expandable title='data object'>
    <ResponseField name='accountNumberId' type='string'>
      Unique identifier of the account number.
    </ResponseField>
    <ResponseField name='accountNumber' type='string'>
      Human-readable account number.
    </ResponseField>
    <ResponseField name='accountName' type='string'>
      Name of the account.
    </ResponseField>
    <ResponseField name='accountOwnerId' type='string'>
      Unique identifier of the account owner.
    </ResponseField>
    <ResponseField name='accountBalance' type='number'>
      Total account balance including all transactions.
    </ResponseField>
    <ResponseField name='availableBalance' type='number'>
      Available balance for transactions (excludes liens and holds).
    </ResponseField>
    <ResponseField name='accountStatus' type='string'>
      Current account status (e.g., `ACTIVATED`, `DEACTIVATED`).
    </ResponseField>
    <ResponseField name='currencyCode' type='string'>
      Account currency code (e.g., `NGN`, `USD`, `GBP`).
    </ResponseField>
    <ResponseField name='overDraftLimit' type='number'>
      Overdraft limit for the account (if applicable).
    </ResponseField>
    <ResponseField name='newOverDraftBalance' type='number'>
      Current overdraft balance used (if applicable).
    </ResponseField>
    <ResponseField name='totalLienAmount' type='number'>
      Total amount placed on lien (frozen/unavailable).
    </ResponseField>
    <ResponseField name='hostBank' type='string'>
      Host bank identifier.
    </ResponseField>
    <ResponseField name='parentAccountId' type='string'>
      Parent account identifier (if this is a sub-account).
    </ResponseField>
    <ResponseField name='accountGroupId' type='string'>
      Account group identifier.
    </ResponseField>
    <ResponseField name='tenancyId' type='string'>
      Tenancy identifier.
    </ResponseField>
    <ResponseField name='actionPerformedByUse' type='string'>
      User who last performed an action on the account.
    </ResponseField>
  </Expandable>
</ResponseField>

```bash cURL
curl --location 'https://{domain}.fincode.software/api/v6/services/ledger/account-balance/123e4567-e89b-12d3-a456-426614174000' \
--header 'X-Auth-Token: YOUR_JWT_ACCESS_TOKEN' \
--header 'x-fapi-auth-date: 2024-01-15T10:30:00Z' \
--header 'x-fapi-customer-ip-address: 192.168.1.1' \
--header 'x-fapi-interaction-id: interaction-12345'
```

```javascript Node.js
const axios = require('axios');

const BASE_URL = 'https://{domain}.fincode.software/api/v6/services/ledger';

async function getAccountBalance(accessToken, accountId) {
  try {
    const response = await axios.get(
      `${BASE_URL}/account-balance/${accountId}`,
      {
        headers: {
          'X-Auth-Token': accessToken,
          'x-fapi-auth-date': new Date().toISOString(),
          'x-fapi-customer-ip-address': '192.168.1.1',
          'x-fapi-interaction-id': `interaction-${Date.now()}`,
        },
      }
    );

    const balance = response.data.data;
    console.log('Account Balance:', balance.accountBalance);
    console.log('Available Balance:', balance.availableBalance);
    console.log('Currency:', balance.currencyCode);
    console.log('Status:', balance.accountStatus);

    return balance;
  } catch (error) {
    console.error(
      'Failed to fetch account balance:',
      error.response?.data || error.message
    );
    throw error;
  }
}

// Usage Example
const accessToken = 'YOUR_JWT_ACCESS_TOKEN';
const accountId = '123e4567-e89b-12d3-a456-426614174000';
getAccountBalance(accessToken, accountId);
```
