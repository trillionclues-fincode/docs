---
title: 'Ledger History by Interaction ID'
api: 'GET https://{domain}.fincode.software/api/v6/services/ledger/account-ledger-history-interaction-id'
description: 'Retrieves ledger entries matching a specific interaction ID (external correlation ID).'
---

Fetches all ledger entries that were processed with a specific interaction ID (external correlation ID). This is useful for retrieving all transactions related to a single interaction or session.

<RequestExample>
  ```bash cURL GET
  [https://{domain}.fincode.software/api/v6/services/ledger/account-ledger-history-interaction-id](https://{domain}.fincode.software/api/v6/services/ledger/account-ledger-history-interaction-id)
  ```
</RequestExample>

## Request Headers

<ParamField header='X-Auth-Token' type='string' required>
  The **JWT Access Token** obtained from the `/login` or `/refresh-token` endpoint.
</ParamField>

<ParamField header='x-fapi-interaction-id' type='string' required>
  The interaction ID (external correlation ID) to search for. This should be a UUID.
</ParamField>

<ParamField header='x-fapi-auth-date' type='string' required>
  The date and time at which the request was initiated (ISO 8601 format).
</ParamField>

<ParamField header='x-fapi-customer-ip-address' type='string' required>
  The IP address of the customer making the request.
</ParamField>

## Response

Returns all ledger entries that match the interaction ID.

<ResponseField name='status' type='string' default='SUCCESS'>
  Overall status of the API request.
</ResponseField>

<ResponseField name='data' type='array'>
  List of ledger entries matching the interaction ID.
  <Expandable title='data array item'>
    <ResponseField name='id' type='string'>
      Unique identifier of the ledger entry.
    </ResponseField>
    <ResponseField name='accountIdentificationNumber' type='string'>
      Account identification number ID.
    </ResponseField>
    <ResponseField name='transactionAmount' type='number'>
      Amount of the transaction.
    </ResponseField>
    <ResponseField name='previousAccountBalance' type='number'>
      Account balance before this transaction.
    </ResponseField>
    <ResponseField name='newBalance' type='number'>
      Account balance after this transaction.
    </ResponseField>
    <ResponseField name='currencyCode' type='string'>
      Currency code of the transaction.
    </ResponseField>
    <ResponseField name='ledgerType' type='string'>
      Type of ledger entry (e.g., `DEBIT`, `CREDIT`).
    </ResponseField>
    <ResponseField name='status' type='string'>
      Status of the transaction.
    </ResponseField>
    <ResponseField name='paymentReference' type='string'>
      Payment reference or transaction reference.
    </ResponseField>
    <ResponseField name='executionDate' type='string'>
      Date and time when the transaction was executed.
    </ResponseField>
    <ResponseField name='idempotencykey' type='string'>
      Idempotency key used for this transaction.
    </ResponseField>
    <ResponseField name='externalCorrelationId' type='string'>
      External correlation ID (interaction ID) for tracking.
    </ResponseField>
    <ResponseField name='internalCorrelationId' type='string'>
      Internal correlation ID for tracking.
    </ResponseField>
  </Expandable>
</ResponseField>

## Code Examples

<CodeGroup>

```bash cURL
curl --location 'https://{domain}.fincode.software/api/v6/services/ledger/account-ledger-history-interaction-id' \
--header 'X-Auth-Token: YOUR_JWT_ACCESS_TOKEN' \
--header 'x-fapi-interaction-id: 123e4567-e89b-12d3-a456-426614174000' \
--header 'x-fapi-auth-date: 2024-01-15T10:30:00Z' \
--header 'x-fapi-customer-ip-address: 192.168.1.1'
```

```javascript Node.js
const axios = require('axios');

const BASE_URL = 'https://{domain}.fincode.software/api/v6/services/ledger';

async function getLedgerHistoryByInteractionId(accessToken, interactionId) {
  try {
    const response = await axios.get(
      `${BASE_URL}/account-ledger-history-interaction-id`,
      {
        headers: {
          'X-Auth-Token': accessToken,
          'x-fapi-interaction-id': interactionId,
          'x-fapi-auth-date': new Date().toISOString(),
          'x-fapi-customer-ip-address': '192.168.1.1',
        },
      }
    );

    console.log(`Found ${response.data.data.length} transactions with interaction ID ${interactionId}:`);
    response.data.data.forEach((entry) => {
      console.log(
        `${entry.executionDate}: ${entry.ledgerType} ` +
        `${entry.transactionAmount} (Balance: ${entry.newBalance})`
      );
    });

    return response.data.data;
  } catch (error) {
    console.error(
      'Failed to fetch ledger history:',
      error.response?.data || error.message
    );
    throw error;
  }
}

// Usage Example
const accessToken = 'YOUR_JWT_ACCESS_TOKEN';
const interactionId = '123e4567-e89b-12d3-a456-426614174000';
getLedgerHistoryByInteractionId(accessToken, interactionId);
```
