---
title: 'Create User Wallet'
api: 'POST https://test-investment-products-service.fincode.software/api/v1/user/wallet/create-wallet'
description: 'Creates a new wallet account for a customer to hold investment funds.'
---

Creates a new wallet account for a customer. This wallet will be used to fund investments and receive returns.

<RequestExample>
  ```bash cURL
  curl --request POST 'https://test-investment-products-service.fincode.software/api/v1/user/wallet/create-wallet' \
  --header 'Authorization: Bearer <token>' \
  --header 'Content-Type: application/json' \
  --data '{
    "currency": "NGN",
    "accountName": "John Doe Investment Wallet",
    "customerId": "8ecca811-607f-44e1-a57e-cd856c50437d"
  }'
  ```
</RequestExample>

## Request Headers

<ParamField header='X-Auth-Token' type='string' required>
  The **JWT Access Token** obtained from the `/login` or `/refresh-token` endpoint.
</ParamField>

<ParamField header='x-idempotency-key' type='string' required>
  Unique idempotency key for the request to prevent duplicate processing.
</ParamField>

<ParamField header='x-fapi-auth-date' type='string' required>
  The date and time at which the request was initiated (ISO 8601 format).
</ParamField>

<ParamField header='x-fapi-customer-ip-address' type='string' required>
  The IP address of the customer making the request.
</ParamField>

<ParamField header='x-fapi-interaction-id' type='string' required>
  Unique identifier for the interaction/session.
</ParamField>

<ParamField header='Content-Type' type='string' required default='application/json'>
  Must be `application/json`.
</ParamField>

## Request Body

<ParamField body='currency' type='string' required>
  Currency code for the wallet (e.g., `NGN`, `USD`, `GBP`).
</ParamField>

<ParamField body='accountName' type='string' required>
  Name for the wallet account.
</ParamField>

<ParamField body='customerId' type='string' required>
  The unique identifier of the customer who owns this wallet.
</ParamField>

## Response

Returns the created wallet details.

<ResponseField name='status' type='string' default='SUCCESS'>
  Overall status of the API request.
</ResponseField>

<ResponseField name='data' type='object'>
  Wallet creation details.
  <Expandable title='data object'>
    <ResponseField name='id' type='string'>
      Unique identifier for the created wallet.
    </ResponseField>
    <ResponseField name='createdDate' type='string'>
      Date and time when the wallet was created.
    </ResponseField>
  </Expandable>
</ResponseField>

```javascript Node.js
const axios = require('axios');

const BASE_URL = 'https://test-investment-products-service.fincode.software/api/v1/user/wallet';

async function createWallet(accessToken, walletData) {
  try {
    const response = await axios.post(
      `${BASE_URL}/create-wallet`,
      {
        currency: walletData.currency,
        accountName: walletData.accountName,
        customerId: walletData.customerId,
      },
      {
        headers: {
          'X-Auth-Token': accessToken,
          'x-idempotency-key': `key-${Date.now()}`,
          'x-fapi-auth-date': new Date().toISOString(),
          'x-fapi-customer-ip-address': '192.168.1.1',
          'x-fapi-interaction-id': `interaction-${Date.now()}`,
          'Content-Type': 'application/json',
        },
      }
    );

    const wallet = response.data.data;
    console.log('Wallet created successfully!');
    console.log('Wallet ID:', wallet.id);
    console.log('Created Date:', wallet.createdDate);
    return wallet;
  } catch (error) {
    console.error(
      'Failed to create wallet:',
      error.response?.data || error.message
    );
    throw error;
  }
}

// Usage Example
const accessToken = 'YOUR_JWT_ACCESS_TOKEN';
createWallet(accessToken, {
  currency: 'NGN',
  accountName: 'John Doe Investment Wallet',
  customerId: '123e4567-e89b-12d3-a456-426614174000',
});
```
