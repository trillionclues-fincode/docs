---
title: 'List Customer Portfolios'
openapi: 'GET /portfolios/{customer-id}'
description: 'Retrieves a paginated list of all investment portfolios for a customer.'
---

Fetches all investment portfolios associated with a customer, including details about each investment such as amount invested, total payable, and product information.

## Request Headers

<ParamField header='X-Auth-Token' type='string' required>
  The **JWT Access Token** obtained from the `/login` or `/refresh-token` endpoint.
</ParamField>

<ParamField header='x-idempotency-key' type='string' required>
  Unique idempotency key for the request to prevent duplicate processing.
</ParamField>

<ParamField header='x-fapi-auth-date' type='string' required>
  The date and time at which the request was initiated (ISO 8601 format).
</ParamField>

<ParamField header='x-fapi-customer-ip-address' type='string' required>
  The IP address of the customer making the request.
</ParamField>

<ParamField header='x-fapi-interaction-id' type='string' required>
  Unique identifier for the interaction/session.
</ParamField>

## Path Parameters

<ParamField path='customer-id' type='string' required>
  The unique identifier (UUID) of the customer whose portfolios to retrieve.
</ParamField>

## Query Parameters

<ParamField query='page' type='number' default='1'>
  Page number for pagination (starts from 1).
</ParamField>

<ParamField query='size' type='number' default='10'>
  Number of items per page (maximum recommended: 50).
</ParamField>

```javascript Node.js
const axios = require('axios');

const BASE_URL = 'https://test-investment-products-service.fincode.software/api/v1/user';

async function listPortfolios(accessToken, customerId, page = 1, size = 10) {
  try {
    const response = await axios.get(
      `${BASE_URL}/portfolios/${customerId}`,
      {
        params: {
          page,
          size,
        },
        headers: {
          'X-Auth-Token': accessToken,
          'x-idempotency-key': `key-${Date.now()}`,
          'x-fapi-auth-date': new Date().toISOString(),
          'x-fapi-customer-ip-address': '192.168.1.1',
          'x-fapi-interaction-id': `interaction-${Date.now()}`,
        },
      }
    );

    const portfolios = response.data.data;
    console.log(`Found ${portfolios.length} portfolios`);
    
    portfolios.forEach((portfolio) => {
      console.log(
        `- ${portfolio.productName}: ${portfolio.amountInvested} ` +
        `(APY: ${portfolio.annualPercentageYield}%)`
      );
    });

    return portfolios;
  } catch (error) {
    console.error(
      'Failed to fetch portfolios:',
      error.response?.data || error.message
    );
    throw error;
  }
}

```
